	if = {
		limit = {
			THIS = { exists = yes }
		}
		if = {
			limit = {
				is_subject = yes
			}
			custom_trigger_tooltip = {
				tooltip = before_declare_war2_tt
				OVERLORD = {
					not = {
						has_non_aggression_pact_with = ROOT
						is_in_faction_with = ROOT
					}
				}
			}
			else = {
				custom_trigger_tooltip = {
					tooltip = before_declare_war1_tt
					not = {
						has_non_aggression_pact_with = ROOT
						is_in_faction_with = ROOT
					}
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = before_declare_war3_tt
			if = {
				limit = {
					is_subject = yes
				}
				OVERLORD = {
					or = {
						is_neighbor_of = ROOT
						any_neighbor_country = {
							is_subject_of = ROOT
						}
						any_neighbor_country = {
							is_in_faction_with = ROOT
						}

							is_neighbor_of = ROOT
						}

							any_neighbor_country = {
								is_subject_of = ROOT
							}
						}
						any_allied_country = {
							any_neighbor_country = {
								is_in_faction_with = ROOT
							}
						}

						any_country = {
							is_subject_of = PREV
							is_neighbor_of = ROOT
						}
						any_country = {
							is_subject_of = PREV
							any_neighbor_country = {
								is_subject_of = ROOT
							}
						}
						any_country = {
							is_subject_of = PREV
							any_neighbor_country = {
								is_in_faction_with = ROOT
							}
						}

						any_country = {

							is_neighbor_of = ROOT
						}

							is_in_faction_with = PREV

								is_subject_of = ROOT
							}
						}
						any_country = {

							}
						}

							is_subject_of = ROOT
						}
